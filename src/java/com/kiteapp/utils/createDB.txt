DROP TABLE "USERDATA";

CREATE TABLE "USERDATA"
(
"USER_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"EMAIL" VARCHAR(50), 
"PASSWORD" VARCHAR(25), 
"FNAME" VARCHAR(50),   
"LNAME" VARCHAR(50),
"USERTYPE" VARCHAR(10)
);

DROP TABLE "PRODUCT";

CREATE TABLE "PRODUCT"
(
"PRODUCT_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"NAME" VARCHAR(50),
"COLOUR" VARCHAR(50),
"SHAPE" VARCHAR(50),
"MATERIAL" VARCHAR(50),
"LEVEL" VARCHAR(50),
"COST" DECIMAL (4,2)
);
